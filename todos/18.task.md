# Session으로 대화 이력 관리

## 세션 관리 방식

- JSONL 형식으로 한 줄에 메시지 하나씩 저장

```jsonl
{"timestamp":"2024-01-15T10:00:00Z","message":{"role":"user","content":"README.md 읽어줘"}}
{"timestamp":"2024-01-15T10:00:05Z","message":{"role":"assistant","content":"파일내용입니다..."}}
```

## 구현 전략

- 아래의 구현절차를 순차적으로 적용
- kotlin 언어로 개발된 것들을 java로 변환하여 프로젝트에 적용

## 구현 절차

### Ste1: 인터페이스 정의 및 저장 형식 정의

```ConversationHistoryStorage.kt
package ai.theunderdog.storage

import ai.koog.prompt.message.Message

interface ConversationHistoryStorage {
  suspend fun addConversation(userMessage: String, assistantMessage: String)
  suspend fun getHistory(): List<Message>
}

@Serializable
data class JsonlEntry(
  val timeStamp: Instant,
  val message: Message
)
```

### Step2: JSONL 구현

```koogagen.storage.JsonlConversationHistoryStorage.kt
// Koog는 JVMFileSystemProvider 를 통해 파일 시스템을 추상화해요.
// init 블록에서 세션 디렉토리가 없으면 자동으로 생성해요. 이렇게 하면 처음 실행할 때 디렉토리를 따로 만들어줄 필요가 없어요.
class JsonlConversationHistoryStorage(
  private val fs: JVMFileSystemProvider.ReadWrite,
  private val sessionDir: java.nio.file.Path,
  private val json: Json = Json {
    ignoreUnknownKeys = true
    prettyPrint = false
  },
) : ConversationHistoryStorage {
  private val historyFile: java.nio.file.Path
    get() = fs.joinPath(sessionDir, "session.jsonl")

  init {
    // 세션 디렉토리 생성
    kotlinx.coroutines.runBlocking {
      if (!fs.exists(sessionDir)) {
        fs.createDirectory(sessionDir)
      }
    }
  }
}  
```

### Step 3: 대화 저장 구현

```addConversation 메소드 추가
/* 핵심 로직
1. 기존 파일이 있으면 읽어오기
2. user 메시지와 assistant 메시지를 JsonlEntry로 변환
3. JSON으로 직렬화해서 한 줄씩 추가
4. 파일에 다시 쓰기
*/
override suspend fun addConversation(
  userMessage: String,
  assistantMessage: String
) {
    val existingContent = if (fs.exists(historyFile)) {
      fs.readText(historyFile)
    } else ""
    
    val userEntry = JsonlEntry(
      timeStamp = now(),
      message = Message.User(userMessage, RequestMetaInfo.create(Clock.System))
    )
    val assistantEntry = JsonlEntry(
      timeStamp = now(),
      message = Message.Assistant(assistantMessage, ResponseMetaInfo.create(Clock.System))
    )
    val newLines = listOf(userEntry, assistantEntry)
      .joinToString("\n") { entry ->
        json.encodeToString(JsonlEntry.serializer(), entry)
      }

    val updatedContent = if (existingContent.isEmpty()) {
      newLines
    } else {
      existingContent + "\n" + newLines
    }
    fs.writeText(historyFile, updatedContent)
}
```

### Step 4: 이력 불러오기

```getHistory 메서드
// 파일을 읽어서 줄 단위로 파싱
// mapNotNull 을 쓰는 이유 (파일이 손상됐거나 형식이 맞지 않는 줄이 있을 수 있음. 그런 줄은 건너뛰고 나머지만 불러오도록 하기 위함)
override suspend fun getHistory(): List<Message> {
  if (!fs.exists(historyFile)) {
    return emptyList()
  }
  val content = fs.readText(historyFile)
  return content.lines()
    .filter { it.isNotBlank() }
    .mapNotNull { line ->
      try {
        val entry = json.decodeFromString<JsonlEntry>(line)
        entry.message
      } catch (e: Exception) {
        System.err.println("해당 줄을 구문 분석하는데 실패했습니다. $line" )
        null
      }
    }
}
```

### Step 5: CodingAgent에 통합 (2/2)

```
suspend fun chat(userMessage: String): String {
  val history = conversationHistoryStorage.getHistory()
  val system = buildSystemPromptWithHistory(history)
  
  val agent = AIAgent(
    promptExecutor = executor,
    systemPrompt = system,
    llmModel = AnthropicModels.Opus_4_6,
    toolRegistry = toolRegistry,
    maxIterations = 10
  )

  val assistantMessage = agent.run(userMessage)
  conversationHistoryStorage.addConversation(
    userMessage = userMessage,
    assistantMessage = assistantMessage,
  )
  return assistantMessage
}

private fun buildSystemPromptWithHistory(history: List<Message>): String {
  if (history.isEmpty()) {
    return systemPrompt
  }
  
  return buildString {
    appendLine("# System Prompt")
    appendLine(systemPrompt)
    appendLine()
    appendLine("# Conversation History")
    history.forEach { message ->
        when (message) {
          is Message.User -> appendLine("User: ${message.content}")
          is Message.Assistant -> appendLine("Assistant: ${message.content}")
          else -> {}
        }
    }
    appendLine()
    appendLine("위의 맥락을 바탕으로 대화를 이어가 주세요.")
  }
}
```
